var fibsFizzBuzz = function(n){
  if (n === 1){
    return [1]
  }
  let arr = [1,1]
  for(let i = 2; i<n; i++){
    arr.push(arr[i - 1] + arr[i - 2])
  }
  return arr.map(el => el%15 === 0? 'FizzBuzz':
                 el%3 === 0? 'Fizz':
                 el%5 === 0? 'Buzz':
                 el)
}
